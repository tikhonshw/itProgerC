git ini - для инициализации проекта
git config --global user.name "tikhonshw" - установка глобального имени
git config --global user.name - просмотр имени
git config --global user.email "tikhonsh@gmail.com" - установка почты пользователя
git add - добавление файлов в локальный репозиторий (режим ожидания)
git status - просмотр статус файлов на загрузку
git rm --cached main.cpp - удаление файла на загрузку (указывать полный путь, полный путь смотрим по git status)
git rm --cached . - удалиь все файлы из ожидания на загрузку
git add *.cpp *.h - загрузка в ожидание по маске
git commit - добавление файлов из иста ожидания в локальный репозиторий, с ключом -m "добавляем описание"

Удаление загрузки
git reset --hard HEAD~1 - полное удаление коментарие + изменение кода
git reset --soft HEAD~1 - удаление коментария БЕЗ правки кода
